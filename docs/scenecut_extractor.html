112    def calculate_scenecuts(
113        self, threshold: float = DEFAULT_THRESHOLD, progress: bool = False
114    ) -> None:
115        """
116        Calculate scene cuts with ffmpeg.
117
118        Args:
119            threshold (float): Threshold (between 0 and 1)
120            progress (bool): Show a progress bar on stderr
121        """
122        if not (0 <= threshold <= 1):
123            raise RuntimeError("Threshold must be between 0 and 1")
124
125        temp_dir = tempfile.mkdtemp()
126        temp_file_name = posixpath.join(
127            temp_dir, "scenecut-extractor-" + os.path.basename(self.input_file) + ".txt"
128        )
129
130        logger.debug("Writing to temp file: " + temp_file_name)
131
132        try:
133            cmd = [
134                "ffmpeg",
135                "-nostdin",
136                "-loglevel",
137                "error",
138                "-y",
139                "-i",
140                self.input_file,
141                "-vf",
142                "select=gte(scene\,0),metadata=print:file="
143                + win_path_check(temp_file_name),
144                "-an",
145                "-f",
146                "null",
147                os.devnull,
148            ]
149
150            logger.debug(
151                "Running ffmpeg command: " + " ".join([shlex.quote(c) for c in cmd])
152            )
153
154            ff = FfmpegProgress(cmd)
155            if progress:
156                with tqdm(total=100, position=1) as pbar:
157                    for p in ff.run_command_with_progress():
158                        pbar.update(p - pbar.n)
159            else:
160                for _ in ff.run_command_with_progress():
161                    pass
162
163            lines: list[str] = []
164            if os.path.isfile(temp_file_name):
165                with open(temp_file_name, "r") as out_f:
166                    lines = out_f.readlines()
167
168            frames: list[ScenecutInfo] = []
169            last_frame_info: dict = {}
170            for line in lines:
171                line = line.strip()
172                if line.startswith("frame"):
173                    if ret := re.match(
174                        r"frame:(?P<frame>\d+)\s+pts:(?P<pts>[\d\.]+)\s+pts_time:(?P<pts_time>[\d\.]+)",
175                        line,
176                    ):
177                        ret_matches = ret.groupdict()
178                        last_frame_info["frame"] = int(ret_matches["frame"])
179                        last_frame_info["pts"] = float(ret_matches["pts"])
180                        last_frame_info["pts_time"] = float(ret_matches["pts_time"])
181                    else:
182                        raise RuntimeError("Wrongly formatted line: " + line)
183                    continue
184
185                if line.startswith("lavfi.scene_score") and (splits := line.split("=")):
186                    if len(splits):
187                        last_frame_info["score"] = float(splits[1])
188                    else:
189                        raise RuntimeError("Wrongly formatted line: " + line)
190                    frames.append(cast(ScenecutInfo, last_frame_info))
191                    last_frame_info = {}
192
193            self.scenecuts = [f for f in frames if f["score"] >= threshold]
194
195        except Exception as e:
196            raise e
197        finally:
198            if os.path.isfile(temp_file_name):
199                logger.debug("Removing temp file: " + temp_file_name)
200                os.remove(temp_file_name)

208        """
209        Extract all scenes to individual files.
210
211        Args:
212            output_directory (str): Output directory.
213            no_copy (bool, optional): Do not copy the streams, reencode them. Defaults to False.
214            progress (bool, optional): Show progress bar. Defaults to False.

217            raise RuntimeError("No scene cuts calculated yet")

222

225

227            self.cut_part_from_file(

229                output_directory,
230                scene["pts_time"],
231                next_scene["pts_time"],
232                no_copy,
